Connection & Authentication

connect()

async

Establishes connection to the binary options broker. 

async def connect(
    username: str,
    password: str,
    demo: bool = True,
    timeout: int = 30
) -> bool:
    """
    Connect to the binary options platform.
    
    Args:
        username (str): Account username
        password (str): Account password  
        demo (bool): Use demo account (default: True)
        timeout (int): Connection timeout in seconds
        
    Returns:
        bool: True if connection successful
        
    Raises:
        ConnectionError: If connection fails
        AuthenticationError: If credentials invalid
    """

Example:

from binary_options_tools import BinaryOptionsAPI

api = BinaryOptionsAPI()
success = await api.connect("username", "password", demo=True)
if success:
    print("Connected successfully!")
else:
    print("Connection failed")

Trading Methods

place_trade()

async

Places a binary options trade with specified parameters. 

async def place_trade(
    asset: str,
    amount: float,
    direction: str,
    expiry_time: int,
    trade_type: str = "binary"
) -> Dict[str, Any]:
    """
    Place a binary options trade.
    
    Args:
        asset (str): Trading asset (e.g., "EURUSD", "BTCUSD")
        amount (float): Trade amount in account currency
        direction (str): "call" or "put"
        expiry_time (int): Expiry time in seconds
        trade_type (str): Type of trade ("binary", "turbo")
        
    Returns:
        Dict[str, Any]: Trade result with ID, status, and details
        
    Raises:
        InsufficientFundsError: If account balance too low
        InvalidAssetError: If asset not available
        TradingClosedError: If market closed
    """

Example:

# Place a $10 CALL trade on EUR/USD for 1 minute
result = await api.place_trade(
    asset="EURUSD",
    amount=10.0,
    direction="call",
    expiry_time=60
)

print(f"Trade ID: {result['trade_id']}")
print(f"Status: {result['status']}")
print(f"Expected payout: ${result['payout']}")

Event System

on()

method

Registers an event listener for specific events. 

on(eventName, callback) {
    /**
     * Register an event listener.
     * 
     * @param {string} eventName - Name of the event
     * @param {Function} callback - Event handler function
     * 
     * Available events:
     * - 'connected': When connection is established
     * - 'disconnected': When connection is lost
     * - 'trade_placed': When a trade is placed
     * - 'trade_result': When trade result is received
     * - 'price_update': When price data is updated
     * - 'balance_update': When balance changes
     */
}

Example:

// Listen for trade results
api.on('trade_result', (result) => {
    console.log(`Trade ${result.tradeId}: ${result.outcome}`);
    console.log(`Profit/Loss: $${result.profit}`);
});

// Listen for price updates
api.on('price_update', (data) => {
    console.log(`${data.asset}: ${data.price}`);
});

// Listen for connection events
api.on('connected', () => {
    console.log('Successfully connected to broker');
});

api.on('disconnected', () => {
    console.log('Connection lost');
});

pub async fn place_trade(
    &self,
    asset: &str,
    amount: f64,

    //! Place a binary options trade.

}

// Stream prices for multiple assets
